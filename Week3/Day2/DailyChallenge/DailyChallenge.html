// Creamos un array de objetos, cada objeto representa un planeta // con su nombre, color y cantidad de lunas const planets = [ { name: "Mercurio", color: "gray", moons: 0 }, { name: "Venus", color: "yellow", moons: 0 }, { name: "Tierra", color: "blue", moons: 1 }, { name: "Marte", color: "red", moons: 2 }, { name: "Júpiter", color: "orange", moons: 79 }, { name: "Saturno", color: "goldenrod", moons: 82 }, { name: "Urano", color: "lightblue", moons: 27 }, { name: "Neptuno", color: "darkblue", moons: 14 } ]; // Seleccionamos el
donde vamos a agregar los planetas const section = document.querySelector(".listPlanets"); // Recorremos el array de planetas planets.forEach(planet => { // Creamos un div para cada planeta const planetDiv = document.createElement("div"); // Le agregamos la clase "planet" planetDiv.classList.add("planet"); // Le agregamos un color de fondo específico para cada planeta planetDiv.style.backgroundColor = planet.color; // Agregamos el nombre del planeta dentro del div planetDiv.textContent = planet.name; // Recorremos las lunas del planeta (si tiene) for (let i = 0; i < planet.moons; i++) { // Creamos un div para la luna const moonDiv = document.createElement("div"); moonDiv.classList.add("moon"); // Posicionamos la luna aleatoriamente alrededor del planeta // Para simplificar, usamos top y left aleatorios dentro de los límites del planeta const distance = 100 + Math.random() * 50; // distancia desde el centro const angle = Math.random() * 2 * Math.PI; // ángulo en radianes const x = Math.cos(angle) * distance; const y = Math.sin(angle) * distance; moonDiv.style.top = `${y}px`; moonDiv.style.left = `${x}px`; // Agregamos la luna al div del planeta planetDiv.appendChild(moonDiv); } // Finalmente, agregamos el div del planeta al
section.appendChild(planetDiv); });
